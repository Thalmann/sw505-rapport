\lstdefinestyle{csharp}
{
	language=[Sharp]C,
	frame=shadowbox,
	framesep=5pt,
	rulecolor=\color{blue!40!black},
	basicstyle=\ttfamily\footnotesize,
	numbers=left,
	numberstyle=\tiny,
	numberfirstline=true,
	numberblanklines=false,
	stepnumber=3,
	numbersep=9pt
}

{
	language=[Sharp]F,
	frame=lr,
	rulecolor=\color{blue!80!black}
}
